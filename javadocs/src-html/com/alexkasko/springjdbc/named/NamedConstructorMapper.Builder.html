001    package com.alexkasko.springjdbc.named;
002    
003    import org.springframework.jdbc.core.ColumnMapRowMapper;
004    import org.springframework.jdbc.core.RowMapper;
005    
006    import java.util.HashMap;
007    import java.util.Map;
008    
009    /**
010     * Spring's <a href="http://static.springsource.org/spring/docs/3.0.x/javadoc-api/org/springframework/jdbc/core/RowMapper.html">RowMapper</a>
011     * implementation using {@link NamedConstructorList}. Designed to use with immutable classes (with final fields) -
012     * only constructor invocation is used without any field access.
013     * Constructor arguments must be annotated with JSR330 {@link javax.inject.Named} annotations
014     * (there are other ways to access constructor names in runtime,see
015     * <a href="http://paranamer.codehaus.org/">paranamer project</a>, but we use {@code @Named} annotations only).
016     * Constructors without {@code @Named} annotations on arguments will be ignored.
017     * Constructors with {@code @Named} annotations must have all they arguments annotated with not blank values without duplicates.
018     * All reflection introspection is done on mapper instantiation. {@code @Named} values used in case-insensitive mode
019     * (databases usually use column names so) so all {@code @Named} values and all result set column names must be locale insensitive.
020     * Supports class hierarchies - different subclasses instantiation from single result set based on, use {@link #builder(String)}
021     * method to create subclasses mapper and see {@link NamedConstructorSubclassesMapper} for details.
022     *
023     * @param <T> object type to map data row to
024     * @author alexkasko
025     * Date: 7/6/12
026     * @see NamedConstructor
027     * @see NamedConstructorList
028     * @see NamedConstructorSingleMapper
029     * @see NamedConstructorSubclassesMapper
030     */
031    public abstract class NamedConstructorMapper<T> implements RowMapper<T> {
032        /**
033         * Factory method for single class named constructor mapper
034         *
035         * @param clazz class type to instantiate from row data
036         * @param <T> class type parameter
037         * @return named constructor instance
038         */
039        public static <T> NamedConstructorMapper<T> forClass(Class<T> clazz) {
040            NamedConstructorList<T> fun = NamedConstructorList.forClass(clazz);
041            return new NamedConstructorSingleMapper<T>(fun);
042        }
043    
044        /**
045         * Builder to create subclasses mapper
046         *
047         * @param discColumn discriminator
048         * @param <T> parent class (or interface) type
049         * @return builder instance
050         */
051        public static <T> Builder<T> builder(String discColumn) {
052            return new Builder<T>(discColumn);
053        }
054    
055        /**
056         * Builder class to create subclasses named constructor mapper
057         *
058         * @param <T>
059         */
060        public static class Builder<T> {
061            private final String discColumn;
062            private final HashMap<String, NamedConstructorList<? extends T>> map = new HashMap<String, NamedConstructorList<? extends T>>();
063    
064            /**
065             * Constructor
066             *
067             * @param discColumn discriminator column
068             */
069            public Builder(String discColumn) {
070                this.discColumn = discColumn;
071            }
072    
073            /**
074             * Registers subclass in mapper
075             *
076             * @param discriminator discriminator column value
077             * @param subclass subclass type
078             * @return builder itself
079             */
080            public Builder<T> addSubclass(String discriminator, Class<? extends T> subclass) {
081                map.put(discriminator, NamedConstructorList.forClass(subclass));
082                return this;
083            }
084    
085            /**
086             * Builds mapper
087             *
088             * @return named mapper instance
089             */
090            public NamedConstructorMapper<T> build() {
091                return new NamedConstructorSubclassesMapper<T>(map, discColumn);
092            }
093        }
094    }
