001    package com.alexkasko.springjdbc.named;
002    
003    import org.springframework.jdbc.core.RowMapper;
004    
005    import java.util.HashMap;
006    
007    /**
008     * Spring's <a href="http://static.springsource.org/spring/docs/3.0.x/javadoc-api/org/springframework/jdbc/core/RowMapper.html">RowMapper</a>
009     * implementation using {@link NamedConstructorList}. Designed to use with immutable classes (with final fields) -
010     * only constructor invocation is used without any field access.
011     * Constructor arguments must be annotated with JSR330 {@link javax.inject.Named} annotations
012     * (there are other ways to access constructor names in runtime,see
013     * <a href="http://paranamer.codehaus.org/">paranamer project</a>, but we use {@code @Named} annotations only).
014     * Constructors without {@code @Named} annotations on arguments will be ignored.
015     * Constructors with {@code @Named} annotations must have all they arguments annotated with not blank values without duplicates.
016     * All reflection introspection is done on mapper instantiation. {@code @Named} values used in case-insensitive mode
017     * (databases usually use column names so) so all {@code @Named} values and all result set column names must be locale insensitive.
018     * Supports class hierarchies - different subclasses instantiation from single result set based on, use {@link #builder(String)}
019     * method to create subclasses mapper and see {@link NamedConstructorSubclassesMapper} for details.
020     *
021     * @param <T> object type to map data row to
022     * @author alexkasko
023     * Date: 7/6/12
024     * @see NamedConstructor
025     * @see NamedConstructorList
026     * @see NamedConstructorSingleMapper
027     * @see NamedConstructorSubclassesMapper
028     */
029    public abstract class NamedConstructorMapper<T> implements RowMapper<T> {
030    
031        /**
032         * Static-import-friendly alias for {@link #forClass(Class)}method
033         *
034         * @param clazz class type to instantiate from row data
035         * @param <T> class type parameter
036         * @return named constructor instance
037         */
038        public static <T> NamedConstructorMapper<T> namedConstructorMapper(Class<T> clazz) {
039            return NamedConstructorMapper.forClass(clazz);
040        }
041    
042        /**
043         * Factory method for single class named constructor mapper
044         *
045         * @param clazz class type to instantiate from row data
046         * @param <T> class type parameter
047         * @return named constructor instance
048         */
049        public static <T> NamedConstructorMapper<T> forClass(Class<T> clazz) {
050            NamedConstructorList<T> fun = NamedConstructorList.forClass(clazz);
051            return new NamedConstructorSingleMapper<T>(fun);
052        }
053    
054        /**
055         * Builder to create subclasses mapper
056         *
057         * @param discColumn discriminator
058         * @param <T> parent class (or interface) type
059         * @return builder instance
060         */
061        public static <T> Builder<T> builder(String discColumn) {
062            return new Builder<T>(discColumn);
063        }
064    
065        /**
066         * Builder class to create subclasses named constructor mapper
067         *
068         * @param <T>
069         */
070        public static class Builder<T> {
071            private final String discColumn;
072            private final HashMap<String, NamedConstructorList<? extends T>> map = new HashMap<String, NamedConstructorList<? extends T>>();
073    
074            /**
075             * Constructor
076             *
077             * @param discColumn discriminator column
078             */
079            public Builder(String discColumn) {
080                this.discColumn = discColumn;
081            }
082    
083            /**
084             * Registers subclass in mapper
085             *
086             * @param discriminator discriminator column value
087             * @param subclass subclass type
088             * @return builder itself
089             */
090            public Builder<T> addSubclass(String discriminator, Class<? extends T> subclass) {
091                map.put(discriminator, NamedConstructorList.forClass(subclass));
092                return this;
093            }
094    
095            /**
096             * Builds mapper
097             *
098             * @return named mapper instance
099             */
100            public NamedConstructorMapper<T> build() {
101                return new NamedConstructorSubclassesMapper<T>(map, discColumn);
102            }
103        }
104    }
